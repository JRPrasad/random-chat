// ======================================== // REAL WORKING RANDOM CHAT - SERVER SETUP // ======================================== // server.js - Deploy this on Railway/Render for real functionality const express = require('express'); const http = require('http'); const socketIo = require('socket.io'); const path = require('path'); const app = express(); const server = http.createServer(app); const io = socketIo(server, { cors: { origin: "*", methods: ["GET", "POST"] } }); // Serve the HTML file app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'index.html')); }); // Real user management let waitingUsers = []; let connectedUsers = new Map(); let activeRooms = new Map(); io.on('connection', (socket) => { console.log('User connected:', socket.id); // Send real online count io.emit('userCount', connectedUsers.size + 1); socket.on('findPeer', () => { connectedUsers.set(socket.id, { socket, waiting: true }); // Find another waiting user const waitingUser = waitingUsers.find(userId => connectedUsers.has(userId) && userId !== socket.id ); if (waitingUser) { // Create room and match users const roomId = `room-${Date.now()}`; // Remove from waiting waitingUsers = waitingUsers.filter(id => id !== waitingUser); // Join both to room socket.join(roomId); connectedUsers.get(waitingUser).socket.join(roomId); // Update user status connectedUsers.get(socket.id).waiting = false; connectedUsers.get(waitingUser).waiting = false; activeRooms.set(roomId, [socket.id, waitingUser]); // Notify both users they're matched socket.emit('matched', { roomId, peerId: waitingUser }); connectedUsers.get(waitingUser).socket.emit('matched', { roomId, peerId: socket.id }); console.log(`Matched ${socket.id} with ${waitingUser} in ${roomId}`); } else { // Add to waiting queue waitingUsers.push(socket.id); socket.emit('waiting'); console.log(`${socket.id} added to waiting queue`); } }); // WebRTC signaling socket.on('offer', (data) => { socket.to(data.roomId).emit('offer', { offer: data.offer, peerId: socket.id }); }); socket.on('answer', (data) => { socket.to(data.roomId).emit('answer', { answer: data.answer, peerId: socket.id }); }); socket.on('ice-candidate', (data) => { socket.to(data.roomId).emit('ice-candidate', { candidate: data.candidate, peerId: socket.id }); }); // Chat messages socket.on('message', (data) => { socket.to(data.roomId).emit('message', { message: data.message, peerId: socket.id, timestamp: new Date().toISOString() }); }); // Skip/Next partner socket.on('skip', (roomId) => { if (activeRooms.has(roomId)) { const users = activeRooms.get(roomId); // Notify both users about skip io.to(roomId).emit('partnerSkipped'); // Remove room activeRooms.delete(roomId); // Put both users back in queue users.forEach(userId => { if (connectedUsers.has(userId)) { connectedUsers.get(userId).waiting = true; waitingUsers.push(userId); } }); } }); socket.on('disconnect', () => { console.log('User disconnected:', socket.id); // Remove from waiting queue waitingUsers = waitingUsers.filter(id => id !== socket.id); // Find and close active room for (let [roomId, users] of activeRooms.entries()) { if (users.includes(socket.id)) { const otherUser = users.find(id => id !== socket.id); if (otherUser && connectedUsers.has(otherUser)) { connectedUsers.get(otherUser).socket.emit('partnerDisconnected'); // Put other user back in queue waitingUsers.push(otherUser); connectedUsers.get(otherUser).waiting = true; } activeRooms.delete(roomId); break; } } connectedUsers.delete(socket.id); io.emit('userCount', connectedUsers.size); }); }); const PORT = process.env.PORT || 3000; server.listen(PORT, () => { console.log(`üöÄ Real Random Chat Server running on port ${PORT}`);
